19  import static org.junit.jupiter.api.Assertions.*;
20  
21  import java.util.Collections;
22  import org.junit.jupiter.api.Test;
23  
24  /**
25   * The access control entry test.
26   *
27   * @author Christian Bremer
28   */
29  class AccessControlEntryTest {
30  
31    /**
32     * Gets permission.
33     */
34    @Test
35    void getPermission() {
36      AccessControlEntry ace = new AccessControlEntry();
37      ace.setPermission("read");
38      assertEquals("read", ace.getPermission());
39  
40      ace = AccessControlEntry.builder()
41          .permission("write")
42          .build();
43      assertEquals("write", ace.getPermission());
44  
45      assertNotEquals(ace, null);
46      assertNotEquals(ace, new Object());
47      assertEquals(ace, ace);
48      assertEquals(ace, AccessControlEntry.builder().permission("write").build());
49  
50      assertTrue(ace.toString().contains("write"));
51    }
52  
53    /**
54     * Gets guest.
55     */
56    @Test
57    void getGuest() {
58      AccessControlEntry ace = new AccessControlEntry();
59      ace.setGuest(Boolean.TRUE);
60      assertEquals(Boolean.TRUE, ace.getGuest());
61      ace.setGuest(Boolean.FALSE);
62      assertEquals(Boolean.FALSE, ace.getGuest());
63      ace.setGuest(null);
64      assertEquals(Boolean.FALSE, ace.getGuest());
65  
66      ace = AccessControlEntry.builder()
67          .guest(Boolean.TRUE)
68          .build();
69      assertEquals(Boolean.TRUE, ace.getGuest());
70  
71      assertEquals(ace, AccessControlEntry.builder().guest(Boolean.TRUE).build());
72  
73      assertTrue(ace.toString().contains("true"));
74    }
75  
76    /**
77     * Gets users.
78     */
79    @Test
80    void getUsers() {
81      AccessControlEntry ace = new AccessControlEntry();
82      assertNull(ace.getUsers());
83      ace.setUsers(Collections.singletonList("anna"));
84      assertNotNull(ace.getUsers());
85      assertFalse(ace.getUsers().isEmpty());
86      assertEquals("anna", ace.getUsers().get(0));
87  
88      ace = AccessControlEntry.builder()
89          .users(Collections.singletonList("anna"))
90          .build();
91      assertNotNull(ace.getUsers());
92      assertFalse(ace.getUsers().isEmpty());
93      assertEquals("anna", ace.getUsers().get(0));
94  
95      assertEquals(
96          ace,
97          AccessControlEntry.builder().users(Collections.singletonList("anna")).build());
98  
99      assertTrue(ace.toString().contains("anna"));
100   }

102   /**
103    * Gets roles.
104    */
105   @Test
106   void getRoles() {
107     AccessControlEntry ace = new AccessControlEntry();
108     assertNull(ace.getRoles());
109     ace.setRoles(Collections.singletonList("ROLE_USER"));
110     assertNotNull(ace.getRoles());
111     assertFalse(ace.getRoles().isEmpty());
112     assertEquals("ROLE_USER", ace.getRoles().get(0));
113 
114     ace = AccessControlEntry.builder()
115         .roles(Collections.singletonList("ROLE_USER"))
116         .build();
117     assertNotNull(ace.getRoles());
118     assertFalse(ace.getRoles().isEmpty());
119     assertEquals("ROLE_USER", ace.getRoles().get(0));
120 
121     assertEquals(
122         ace,
123         ace.toBuilder().roles(Collections.singletonList("ROLE_USER")).build());
124 
125     assertTrue(ace.toString().contains("ROLE_USER"));
126   }
